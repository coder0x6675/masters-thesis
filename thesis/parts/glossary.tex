    \item[LIGHT:] Lightning Archive Dataset
    \item[MAE:] Mean Absolute Error

    \item[MESAN:] Meteorological Analysis Model Data Dataset
    \item[MIMO:] Multiple-Input Multiple-Output
    \item[MISO:] Multiple-Input Single-Output
    \item[ML:] Machine Learning
    \item[PC:] Principal Component
    \item[PCA:] Principal Component Analysis
    \item[RNN:] Recurrent Neural Network
    \item[SMHI:] Swedish Meteorological and Hydrological Institute
    \item[SRNN:] Simple Recurrent Neural Network
    \item[TCN:] Temporal Convolutional Neural
    \item[UALF:] Universal ASCII Lightning Format
    \item[VAE:] Variational Autoencoder
    \item[WS:] Wilson Score
